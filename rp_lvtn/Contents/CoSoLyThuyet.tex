\section{Cơ sở lý thuyết}

\subsection{Mô hình Client - Server}
\indent \indent Mô hình Client-Server (Khách - Chủ) là kiến trúc nền tảng phân chia nhiệm vụ giữa bên yêu cầu dịch vụ (Client) và bên cung cấp dịch vụ (Server). Trong hệ thống này, Client đóng vai trò là "cửa ngõ" để người dùng tương tác, trong khi Server giữ vai trò là "trung tâm điều phối" xử lý các tác vụ phức tạp và quản lý tài nguyên. Việc tách biệt này giúp hệ thống vận hành linh hoạt, bảo mật và dễ dàng nâng cấp hơn so với các kiến trúc truyền thống.

    \textbf{Server (Máy chủ)} & Đóng vai trò bộ não điều khiển (sử dụng Node.js và MongoDB). Thành phần này túc trực để tiếp nhận yêu cầu, thực thi các thuật toán nghiệp vụ, kiểm tra quyền truy cập và quản lý dữ liệu lưu trữ một cách an toàn. \\ \hline
    \textbf{Cơ chế giao tiếp} & Là phương thức kết nối thông qua môi trường mạng (Internet/Intranet). Quá trình trao đổi diễn ra theo chu kỳ Request - Response (Yêu cầu - Phản hồi) dựa trên các giao thức chuẩn hóa, đảm bảo dữ liệu đi đúng hướng và toàn vẹn. \\ \hline
\end{longtable}

    \textbf{Dễ mở rộng} & Kiến trúc này cho phép phục vụ đồng thời nhiều người dùng và thuận lợi cho việc nâng cấp tài nguyên phía máy chủ khi lưu lượng truy cập tăng cao, đặc biệt trong các hệ thống đặt vé trực tuyến. \\ \hline
    \textbf{Tăng cường bảo mật} & Server có thể đảm nhiệm việc xác thực, phân quyền, mã hóa và kiểm soát truy cập, từ đó giảm rủi ro phát sinh do thao tác trực tiếp từ phía Client. \\ \hline

\textbf{Mô hình dữ liệu} 
& Tổ chức dữ liệu dựa trên cấu trúc bảng (rows và columns) với các mối quan hệ logic chặt chẽ giữa các thực thể.
& Lưu trữ linh hoạt dưới dạng document, key-value, graph hoặc wide-column, không bắt buộc tuân theo cấu trúc bảng. \\ \hline

\textbf{Đặc tính Schema} 
& Sử dụng Schema cố định (Rigid), yêu cầu thiết kế và chuẩn hóa dữ liệu nghiêm ngặt trước khi đưa vào vận hành.
& Sử dụng Schema động (Flexible), cho phép thay đổi cấu trúc dữ liệu tùy biến mà không cần định nghĩa trước. \\ \hline

\textbf{Mở rộng (Scaling)} 
& Ưu tiên mở rộng theo chiều dọc (Vertical): Tăng cường sức mạnh phần cứng như CPU, RAM cho máy chủ hiện có.
& Thiết kế để mở rộng theo chiều ngang (Horizontal): Phân tán dữ liệu trên nhiều máy chủ phổ thông để chia tải. \\ \hline

\textbf{Tính nhất quán} 
& Đảm bảo tính nhất quán mạnh mẽ (ACID), phù hợp cho các giao dịch cần độ chính xác tuyệt đối như tài chính.
& Thường ưu tiên tính sẵn sàng và nhất quán cuối cùng (Eventual Consistency), tối ưu cho các hệ thống dữ liệu lớn. \\ \hline

\textbf{Truy vấn và Hiệu năng} 
& Sử dụng ngôn ngữ SQL tiêu chuẩn; hiệu quả với các truy vấn phức tạp nhưng có thể chậm khi thực hiện JOIN quá nhiều bảng.
& Tối ưu tốc độ đọc/ghi dữ liệu nhờ mô hình denormalization, giúp truy xuất dữ liệu trực tiếp mà không cần JOIN phức tạp. \\ \hline

\textbf{Hạ tầng lưu trữ} 
& Yêu cầu hạ tầng mạng và lưu trữ chuyên dụng cao cấp (SAN, SSD tốc độ cao) để duy trì hiệu năng ổn định.
& Có khả năng vận hành tốt trên các thiết bị lưu trữ thông thường, giúp tiết kiệm chi phí khi triển khai quy mô lớn. \\ \hline

\textbf{Ví dụ điển hình} 
& MySQL, PostgreSQL, Microsoft SQL Server, Oracle Database.
& MongoDB, Redis, Cassandra, Neo4j, Amazon DynamoDB. \\ \hline
\end{longtable}
\subsection{Xác thực và Phân quyền người dùng (Authentication \& Authorization)}
\indent \indent \textbf{a. Xác thực (Authentication)}\\
\indent Xác thực là quá trình kiểm tra và xác nhận danh tính của người dùng hoặc thiết bị trước khi cho phép truy cập vào hệ thống. Mục tiêu chính là đảm bảo rằng người dùng là chính họ, dựa trên thông tin họ cung cấp như tên đăng nhập, mật khẩu, mã OTP hoặc các phương pháp sinh trắc học. \\
\indent Quá trình xác thực thường bao gồm các bước sau:
\begin{itemize}
    \item[1.] Người dùng nhập thông tin đăng nhập (ví dụ: email và mật khẩu).
    \item[2.] Hệ thống so sánh thông tin này với dữ liệu đã lưu trữ trong cơ sở dữ liệu.
    \item[3.] Nếu thông tin khớp, hệ thống tạo một phiên làm việc (session) hoặc token (như JWT - JSON Web Token) để xác nhận danh tính trong các yêu cầu tiếp theo.
\end{itemize}
\indent \indent Các phương pháp xác thực phổ biến bao gồm:
\begin{itemize}
    \item[-] Xác thực bằng mật khẩu: Phương pháp cơ bản nhưng cần mã hóa (như bcrypt) để tăng tính bảo mật.
    \item[-] Xác thực hai yếu tố (2FA): Kết hợp mật khẩu với mã OTP gửi qua SMS hoặc hệ thống, nâng cao an toàn.
    \item[-] Xác thực bằng token: Sử dụng token không đổi (refresh token) và token truy cập (access token) để quản lý phiên làm việc.
\end{itemize}
\indent \indent \textbf{b. Phân quyền (Authorization)}\\
\indent Phân quyền là quá trình xác định quyền hạn của người dùng đã được xác thực, quyết định họ có thể thực hiện những hành động nào trong hệ thống. \\
\indent Phân quyền thường dựa trên:
\begin{itemize}
    \item[-] Vai trò (Role-Based Access Control - RBAC): Gán quyền theo vai trò, ví dụ quản trị viên có quyền chỉnh sửa sự kiện, trong khi người dùng chỉ xem thông tin.
    \item[-] Quy tắc (Rule-Based Access Control): Áp dụng các quy định cụ thể, như chỉ cho phép mua vé khi sự kiện còn chỗ.
    \item[-] Danh sách kiểm soát truy cập (ACL): Xác định quyền dựa trên danh sách người dùng hoặc nhóm.
\end{itemize}
\indent \indent Sự kết hợp giữa xác thực và phân quyền giúp hệ thống đảm bảo an toàn, bảo mật thông tin và cung cấp trải nghiệm cá nhân hóa cho từng loại người dùng.

\subsection{Hệ thống gợi ý (Recommendation System)}

\indent \indent Trong bối cảnh các hệ thống số ngày càng phát triển, việc cá nhân hóa trải nghiệm người dùng trở thành yếu tố then chốt giúp nâng cao sự hài lòng và giữ chân người dùng. Hệ thống gợi ý (Recommendation System) ra đời nhằm giải quyết bài toán này, bằng cách đề xuất các sản phẩm, dịch vụ hoặc nội dung phù hợp dựa trên dữ liệu hành vi và sở thích của người dùng. \\
\indent Có nhiều phương pháp tiếp cận khác nhau để xây dựng hệ thống gợi ý, trong đó phổ biến nhất bao gồm:

\begin{itemize}
    \item[-] \textbf{Content-Based Filtering}: Dựa trên việc phân tích đặc điểm của sản phẩm (ví dụ: thể loại sự kiện, nội dung) và hồ sơ sở thích của người dùng để đưa ra các đề xuất tương đồng.
    \item[-] \textbf{Collaborative Filtering}: Dựa trên hành vi của tập người dùng (ví dụ: lượt xem, mua vé) để tìm ra những người dùng có xu hướng giống nhau, từ đó đề xuất các nội dung mà những người dùng tương tự đã quan tâm.
    \item[-] \textbf{Hybrid Ranking}: Là phương pháp kết hợp cả Content-Based và Collaborative Filtering nhằm tận dụng ưu điểm của từng phương pháp, giúp cải thiện độ chính xác và tính đa dạng của kết quả gợi ý.
\end{itemize}

\indent \textbf{Lý do lựa chọn Hybrid Ranking trong hệ thống:} Trong phạm vi dự án, việc sử dụng Hybrid Ranking giúp hệ thống không chỉ khai thác thông tin cá nhân hóa từ từng người dùng mà còn tận dụng tri thức từ hành vi của cộng đồng. Phương pháp này đặc biệt hiệu quả trong các trường hợp dữ liệu người dùng còn hạn chế (cold-start) hoặc khi cần đề xuất các sự kiện mới. Nhờ đó, hệ thống có thể cung cấp danh sách gợi ý đa dạng, chính xác và mang tính khám phá cao hơn. \\
\indent \textbf{Kết luận:} Việc áp dụng mô hình Hybrid Ranking đóng vai trò quan trọng trong việc nâng cao chất lượng hệ thống gợi ý, góp phần cải thiện trải nghiệm người dùng và tăng hiệu quả tương tác trong nền tảng đặt vé sự kiện.


\subsection{Thuật toán LightGBM Ranker (LambdaRank - Gradient Boosted Decision Trees for Ranking)}

\indent \indent LightGBM là một framework mã nguồn mở dựa trên thuật toán cây quyết định tăng cường gradient (Gradient Boosting Decision Tree - GBDT), được tối ưu hóa về tốc độ huấn luyện và khả năng xử lý dữ liệu lớn. 
Trong phạm vi đề tài, hệ thống sử dụng biến thể LightGBM Ranker để giải quyết bài toán gợi ý sự kiện dưới dạng bài toán Learning to Rank (LTR).\\
\indent \textbf{a. Nguyên lý hoạt động và Cấu hình LambdaRank}\\
\indent Thay vì dự đoán một giá trị cụ thể, mô hình tập trung vào việc xác định thứ tự ưu tiên của các sự kiện đối với từng người dùng.
\begin{itemize}
    \item[1.] Chiến lược Leaf-wise: LightGBM phát triển cây theo các lá có độ lợi (loss reduction) lớn nhất thay vì phát triển theo từng tầng (level-wise). Điều này giúp mô hình đạt độ chính xác cao hơn và giảm thiểu sai số trong các bài toán xếp hạng phức tạp.
    \item[2.] Hàm mất mát LambdaRank: Mô hình sử dụng LambdaRank để tối ưu hóa trực tiếp các chỉ số đánh giá thứ tự. Thay vì tính toán hàm mất mát dựa trên từng điểm dữ liệu riêng lẻ, LambdaRank tính toán các giá trị "Lambda" (gradient) dựa trên sự thay đổi của chỉ số xếp hạng khi hoán đổi vị trí giữa các cặp mục tiêu (pairwise), giúp mô hình học được thứ tự tương đối hiệu quả hơn.

\end{itemize}

\indent \textbf{b. Phương pháp huấn luyện và Đánh giá mô hình}\\
\indent Để phù hợp với bài toán gợi ý thực tế và quy trình MLOps tự động, các kỹ thuật sau được áp dụng:

\begin{itemize}
    \item[1.] Xây dựng nhãn Interest Score: Biến mục tiêu được rời rạc hóa thành 5 mức độ (từ 0 đến 4) dựa trên hành vi tương tác (click, mua vé). Việc phân cấp này giúp mô hình LambdaRank nhận diện rõ rệt các mức độ ưu tiên, từ đó đưa ra danh sách đề xuất có tính phân hóa cao.
    \item[2.] Chiến lược Group K-Fold: Để đảm bảo tính khách quan khi đánh giá khả năng gợi ý cho người dùng mới (Cold Start), dữ liệu được phân đoạn dựa trên \texttt{user\_id}. Kỹ thuật này đảm bảo toàn bộ hành vi của một người dùng cụ thể chỉ xuất hiện trong một tập (huấn luyện hoặc kiểm thử), giúp mô hình tránh hiện tượng học vẹt và đánh giá đúng năng lực tổng quát hóa.
    \item[3.] Độ đo tối ưu NDCG (Normalized Discounted Cumulative Gain): Mô hình tập trung tối ưu chỉ số NDCG@15. Đây là độ đo quan trọng trong hệ thống xếp hạng, ưu tiên các sự kiện có độ liên quan cao nhất xuất hiện ở các vị trí đầu tiên trong danh sách 15 kết quả trả về cho người dùng.
\end{itemize}

\indent \textbf{c. Tối ưu hóa triển khai với định dạng ONNX}\\
\indent Nhằm đáp ứng yêu cầu về hiệu suất trong môi trường vận hành thực tế (Production), mô hình LightGBM sau khi huấn luyện được chuyển đổi sang định dạng ONNX (Open Neural Network Exchange). Việc này mang lại hai lợi ích chính:
\begin{itemize}
    \item[1.] Giảm độ trễ (Inference Latency): Tối ưu hóa tốc độ phản hồi của Recommendation Server khi nhận yêu cầu từ Web Server.
    \item[2.] Tính linh hoạt: Loại bỏ sự phụ thuộc vào thư viện LightGBM gốc, giúp hệ thống gọn nhẹ và dễ dàng tích hợp vào quy trình MLOps tự động kết nối giữa GitHub Actions và Hugging Face Spaces.
\end{itemize}

\subsection{Quy trình MLOps và Tự động hóa (Automated Machine Learning Pipeline)}

\indent \indent Trong các hệ thống ứng dụng trí tuệ nhân tạo hiện đại, việc xây dựng mô hình machine learning chỉ là một phần của toàn bộ vòng đời sản phẩm. Để đảm bảo mô hình hoạt động ổn định, có thể mở rộng và dễ dàng bảo trì trong môi trường thực tế, quy trình MLOps (Machine Learning Operations) được áp dụng nhằm tự động hóa và chuẩn hóa toàn bộ pipeline phát triển mô hình.\\
\indent Quy trình MLOps thường bao gồm các giai đoạn chính sau:

\begin{itemize}
    \item[1.] \textbf{Data Collection and Preprocessing}: Thu thập dữ liệu từ nhiều nguồn khác nhau, sau đó thực hiện các bước làm sạch, chuẩn hóa và biến đổi dữ liệu để phục vụ cho quá trình huấn luyện.
    \item[2.] \textbf{Model Training}: Sử dụng dữ liệu đã xử lý để huấn luyện mô hình machine learning với các thuật toán phù hợp.
    \item[3.] \textbf{Model Evaluation}: Đánh giá hiệu suất mô hình thông qua các tập kiểm thử và các chỉ số như Accuracy, Precision, Recall, F1-score hoặc NDCG đối với bài toán xếp hạng.
    \item[4.] \textbf{Model Deployment}: Triển khai mô hình vào môi trường thực tế dưới dạng dịch vụ (API hoặc Model Server) để phục vụ người dùng hoặc hệ thống khác.
    \item[5.] \textbf{Monitoring and Maintenance}: Theo dõi hiệu suất mô hình sau khi triển khai, phát hiện suy giảm chất lượng (model drift) và thực hiện cập nhật hoặc huấn luyện lại khi cần thiết.
\end{itemize}

\indent \textbf{Tự động hóa quy trình MLOps:} Việc tích hợp các công cụ CI/CD như GitHub Actions cho phép tự động hóa toàn bộ pipeline, từ kiểm tra mã nguồn, huấn luyện mô hình, đánh giá đến triển khai. Điều này giúp giảm thiểu sai sót do con người, rút ngắn thời gian phát triển và đảm bảo mô hình luôn được cập nhật với dữ liệu mới nhất.\\
\indent \textbf{Kết luận:} Áp dụng MLOps và tự động hóa pipeline không chỉ nâng cao hiệu quả phát triển mô hình machine learning mà còn đảm bảo tính ổn định, khả năng mở rộng và duy trì lâu dài của hệ thống trong môi trường thực tế.

    \item[-] \textbf{Hiệu năng truy vấn tốt:} Mô hình document giúp giảm phụ thuộc vào JOIN phức tạp, từ đó cải thiện tốc độ đọc/ghi trong nhiều tác vụ thực tế.
    \item[-] \textbf{Triển khai nhanh:} Cú pháp truy vấn trực quan, gần với cấu trúc dữ liệu trong code, giúp rút ngắn thời gian phát triển và bảo trì.

\indent \indent Với hệ thống đặt vé, nhóm xem xử lý ảnh là thành phần quan trọng để đảm bảo trải nghiệm người dùng và hiệu năng tải trang. Vì vậy, đề tài kết hợp \textbf{Multer} và \textbf{Cloudinary} để tạo pipeline tải lên - lưu trữ - tối ưu hóa ảnh gọn và ổn định.
\begin{itemize}
    \item[-] \textbf{Multer:} Middleware Node.js để nhận file dạng multipart/form-data, trích xuất tệp từ request và lưu tạm trước khi xử lý tiếp.
    \item[-] \textbf{Cloudinary:} Dịch vụ đám mây để lưu trữ, phân phối và tự động tối ưu ảnh (định dạng, kích thước, chất lượng) mà không làm nặng backend.
\end{itemize}

\indent \textbf{Lợi ích chính của cách kết hợp này:}
\begin{itemize} 
    \item[-] \textbf{Nhanh và chính xác ở bước upload:} Multer chuẩn hóa quy trình nhận file từ người dùng.
    \item[-] \textbf{Giảm tải hệ thống:} Cloudinary tách lưu trữ ảnh khỏi server ứng dụng, thuận lợi khi mở rộng.
    \item[-] \textbf{Tối ưu trải nghiệm truy cập:} Ảnh được tối ưu theo thiết bị và băng thông, giúp trang tải mượt hơn.
    \item[-] \textbf{Dễ triển khai:} Cả hai công cụ có tài liệu tốt, rút ngắn thời gian phát triển và bảo trì.
\end{itemize}

\indent Tóm lại, Multer + Cloudinary giúp nhóm xây dựng luồng xử lý ảnh rõ ràng, hiệu quả và dễ mở rộng, đồng thời giữ chất lượng trải nghiệm người dùng ở mức ổn định.

    \item[-] \textbf{Hỗ trợ tự động hóa:} Phối hợp tốt với CI/CD (như GitHub Actions) để cập nhật mô hình liên tục.
\end{itemize}

\indent Tóm lại, việc kết hợp Datasets và Spaces giúp nhóm duy trì pipeline ML gọn, dễ mở rộng và sẵn sàng đưa mô hình vào vận hành thực tế.

\subsubsection{ONNX (Open Neural Network Exchange): Định dạng tối ưu hóa hiệu suất dự đoán}

\begin{figure}[H]
    \centering
    \includegraphics[width=0.25\textwidth]{Images/ONNX.png}
    \vspace{15pt}
    \caption[Định dạng ONNX (Open Neural Network Exchange)]{Nguồn ảnh tham khảo từ tài liệu tham khảo \cite{onnx}.}
\end{figure}
\indent \indent Một trong những thách thức lớn nhất khi đưa trí tuệ nhân tạo vào thực tiễn là sự khác biệt giữa môi trường huấn luyện và môi trường vận hành. ONNX (Open Neural Network Exchange) ra đời như một chuẩn định dạng trung gian mã nguồn mở, cho phép các mô hình học máy có thể "giao tiếp" và chuyển đổi linh hoạt giữa các framework khác nhau mà không bị bó hẹp trong một hệ sinh thái duy nhất.

Bằng cách sử dụng ONNX, quy trình phát triển được tách biệt hóa: mô hình có thể được xây dựng trên các công cụ chuyên về huấn luyện (như LightGBM, PyTorch) nhưng lại được thực thi (inference) trên các môi trường tối ưu hóa hiệu năng thông qua ONNX Runtime.

\begin{itemize}
    \item[-] \textbf{Chuẩn hóa cấu trúc mô hình:} ONNX cung cấp một đồ thị tính toán định dạng chung, giúp đồng nhất các toán tử và cấu trúc dữ liệu từ nhiều nguồn khác nhau về một tiêu chuẩn duy nhất.
    \item[-] \textbf{Đẩy mạnh tốc độ phản hồi:} Khi được chuyển đổi sang ONNX, mô hình có khả năng tận dụng tối đa sức mạnh phần cứng thông qua các kỹ thuật tối ưu đồ thị, từ đó giảm thiểu độ trễ (latency) và tiết kiệm tài nguyên máy chủ.
\end{itemize}

\indent \textbf{Vai trò cốt lõi trong quy trình MLOps:}
\begin{itemize}
    \item[-] \textbf{Xóa bỏ rào cản nền tảng:} Cho phép triển khai mô hình lên các môi trường production đa dạng (từ Cloud đến Edge) mà không cần cài đặt lại toàn bộ thư viện huấn luyện ban đầu vốn rất cồng kềnh.
    \item[-] \textbf{Tăng cường tính ổn định:} Việc sử dụng một định dạng tĩnh như ONNX giúp quá trình dự đoán diễn ra nhất quán, hạn chế các lỗi phát sinh do khác biệt phiên bản giữa các framework.
    \item[-] \textbf{Tối ưu hóa Pipeline Recommendation:} Cho phép hệ thống gợi ý thực hiện các phép tính toán phức tạp trong thời gian thực, đáp ứng ngay lập tức nhu cầu của người dùng tại thời điểm truy cập.
\end{itemize}

\indent \indent Trong phạm vi dự án, mô hình sau khi đạt độ chính xác mục tiêu sẽ được đóng gói dưới dạng tệp tin \texttt{.onnx}. Tệp tin này sau đó được tích hợp trực tiếp vào Recommendation Server, đóng vai trò là "trái tim" cung cấp các kết quả dự đoán nhanh chóng, ổn định và sẵn sàng phục vụ lượng truy cập lớn.

\indent \textbf{Cơ chế vận hành của Webhooks:}
\begin{itemize}
    \item[-] \textbf{Thông báo sự kiện tức thời:} Ngay khi giao dịch hoàn tất, PayOS sẽ phát một tín hiệu HTTP Request đến máy chủ ứng dụng, loại bỏ việc phải liên tục truy vấn (polling) trạng thái thủ công.
    \item[-] \textbf{Tự động hóa trạng thái đơn hàng:} Hệ thống Backend căn cứ vào dữ liệu từ Webhook để chuyển đổi trạng thái vé hoặc dịch vụ (từ chờ thanh toán sang đã xác nhận) một cách chính xác.
    \item[-] \textbf{Đảm bảo tính vẹn toàn dữ liệu:} Cơ chế này hạn chế tối đa các sai sót do ngắt kết nối giữa trình duyệt và máy chủ, đảm bảo thông tin trên Dashboard của người bán luôn khớp với thực tế thanh toán.
\end{itemize}

\indent \textbf{Lợi ích triển khai thực tế:}
\begin{itemize}
    \item[-] \textbf{Tối ưu quy trình phát triển:} API chuẩn hóa giúp rút ngắn thời gian tích hợp các phương thức thanh toán phức tạp vào hệ thống hiện có.
    \item[-] \textbf{Nâng cao tính tương tác:} Người dùng nhận được kết quả phản hồi gần như ngay lập tức sau khi quét mã QR hoặc thực hiện chuyển khoản.
    \item[-] \textbf{Vận hành ổn định 24/7:} Webhooks cho phép hệ thống xử lý giao dịch tự động tại mọi thời điểm mà không cần sự can thiệp trực tiếp của quản trị viên.
\end{itemize}
