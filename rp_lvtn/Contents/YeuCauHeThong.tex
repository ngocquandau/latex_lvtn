\section{Yêu cầu về hệ thống}
\indent \indent Để xác định và đảm bảo các tính năng ứng với từng vai trò người dùng cùng với khả năng đáp ứng các yêu cầu thực tiễn khi đưa vào hoạt động, chúng tôi cần làm rõ hệ thống cần phải làm gì và vận hành như thế nào để hiện thực hóa các mục tiêu đó. \\
\indent Để thực hiện nhiệm vụ trên, chúng tôi sẽ chi tiết hóa các yêu cầu cụ thể, được chia thành hai nhóm chính: 
\begin{itemize}[leftmargin=3em]
    \item[1. ] \textbf{Yêu cầu Chức năng (Functional Requirements)}: Mô tả các hành vi và chức năng cụ thể mà hệ thống phải có để đáp ứng nhu cầu của người dùng, Ban tổ chức và quản trị viên.
    \item[2. ] \textbf{Yêu cầu Phi chức năng (Non-Functional Requirements)}: Quy định các tiêu chí về chất lượng, hiệu suất, bảo mật và độ tin cậy, đảm bảo hệ thống không chỉ chạy đúng mà còn phải chạy ổn định, an toàn và hiệu quả. 
\end{itemize}
\indent \indent Các yêu cầu này cùng nhau tạo nên bộ khung kỹ thuật toàn diện để định hướng cho quá trình thiết kế, phát triển và kiểm thử hệ thống.

\subsection{Yêu cầu phi chức năng}

\subsubsection{Bảo mật (Security)}

\textbf{Xác thực và Phân quyền}

\begin{itemize}[leftmargin=*, label={--}]
    \item \textbf{JWT (JSON Web Token):} Mọi API bảo vệ đều yêu cầu Bearer Token
    trong header \texttt{Authorization}. Secret key lấy từ biến môi trường
    \texttt{SECRET\_KEY}; thiếu key trả về lỗi HTTP 500 thay vì tiếp tục xử lý.

    \item \textbf{Phân quyền 3 tầng:} Hệ thống phân biệt rõ \texttt{user} /
    \texttt{organizer} / \texttt{admin}. Middleware \texttt{verifyAdmin} chặn truy
    cập trái phép; \texttt{verifyToken} kiểm tra cả trạng thái \texttt{isActive}
    của tài khoản trong cơ sở dữ liệu -- không chỉ kiểm tra chữ ký token.

    \item \textbf{Optional Auth:} Middleware \texttt{optionalAuth} cho phép người
    dùng ẩn danh vẫn sử dụng được một số tính năng (xem sự kiện, tương tác),
    nhưng không ghi nhận interaction.

    \item \textbf{Frontend Protected Route:} Component \texttt{ProtectedRoute.tsx}
    chặn điều hướng sai role ở phía client. Token hết hạn được phát hiện bằng
    cách giải mã payload \texttt{exp} -- tự động xóa token và chuyển hướng về
    trang chủ.
\end{itemize}

\textbf{Mã hóa Dữ liệu Nhạy cảm}

\begin{itemize}[leftmargin=*, label={--}]
    \item \textbf{bcryptjs} với \texttt{saltRounds = 10} để hash password trước
    khi lưu vào MongoDB.

    \item Password \textbf{không bao giờ trả về} trong query nhờ cấu hình
    \texttt{select: false} trên schema.

    \item Người dùng \textbf{không được phép} tự thay đổi trường \texttt{role}
    qua API -- server kiểm tra và xóa field nếu người gửi không phải admin.
\end{itemize}

\textbf{Ngăn chặn Tấn công}

\begin{itemize}[leftmargin=*, label={--}]
    \item \textbf{CORS Whitelist:} Chỉ chấp nhận request từ
    \texttt{https://mticket.vercel.app} và \texttt{http://localhost:3000}.

    \item \textbf{HTML Escaping} trong email service (hàm \texttt{escapeHtml})
    ngăn XSS khi render nội dung động vào HTML email.

    \item \textbf{DOMPurify} (frontend) sanitize nội dung rich-text từ Quill
    editor trước khi render ra giao diện.

    \item \textbf{Cron Job bảo vệ bằng \texttt{X-Cron-Secret}:} Route trigger
    cron từ bên ngoài yêu cầu header bí mật khớp với
    \texttt{process.env.CRON\_SECRET}.

    \item \textbf{Giới hạn query:} Số record trả về tối đa 200 bản ghi
    (\texttt{Math.min(..., 200)}) để tránh tấn công DoS qua over-fetching.

    \item \textbf{Input filter:} Chỉ cho phép các filter hợp lệ
    (\texttt{status}, \texttt{genre}, \texttt{organizer}) trong query --
    không để raw query từ client truyền thẳng vào MongoDB.
\end{itemize}

\textbf{Bảo mật Thanh toán}

\begin{itemize}[leftmargin=*, label={--}]
    \item Tích hợp \textbf{PayOS} với \texttt{checksumKey} riêng để xác minh
    webhook.

    \item QR thanh toán \textbf{hết hạn sau 15 phút} (trường \texttt{expiredAt}),
    đồng bộ với cron job thu hồi vé pending.
\end{itemize}

\subsubsection{Hiệu năng (Performance)}

\textbf{Cơ sở Dữ liệu}

\begin{itemize}[leftmargin=*, label={--}]
    \item \textbf{MongoDB Text Index:} Hỗ trợ full-text search trên sự kiện
    (dùng \texttt{\$text: \{ \$search \}} kết hợp \texttt{textScore}).

    \item \textbf{2dsphere Index:} Trên \texttt{User.location} và
    \texttt{Event.location} cho phép truy vấn địa lý (sự kiện gần nhất theo
    tọa độ người dùng).

    \item \textbf{Index \texttt{createdAt: -1}} trên collection
    \texttt{Statistic} để lấy bản ghi mới nhất nhanh chóng.

    \item \textbf{\texttt{.lean()}:} Sử dụng trong Gemini service và các
    aggregate pipeline để trả về plain JS object, giảm overhead của Mongoose.

    \item \textbf{Aggregate Pipeline:} Dữ liệu interaction và thống kê được
    tính toán trực tiếp trong MongoDB thay vì load toàn bộ về Node.js.
\end{itemize}

\textbf{Caching}

\begin{itemize}[leftmargin=*, label={--}]
    \item \textbf{Logo caching:} Biến \texttt{cachedLogoAttachment} và
    \texttt{watermarkLogoDataUri} cache file ảnh logo vào bộ nhớ -- chỉ đọc
    disk một lần duy nhất.

    \item \textbf{Redis} (dependency \texttt{redis} \^{}5.10.0): Sẵn sàng làm
    caching layer / session store cho toàn hệ thống.
\end{itemize}

\textbf{Phân trang và Giới hạn Tải}

\begin{itemize}[leftmargin=*, label={--}]
    \item Phân trang chuẩn với các tham số: \texttt{page}, \texttt{limit},
    \texttt{direction}, \texttt{sortField}.

    \item \textbf{Cursor-based pagination} hỗ trợ thêm để tránh \texttt{skip()}
    chậm trên collection lớn.

    \item Chatbot AI chỉ load \textbf{tối đa 30 sự kiện} published gần nhất,
    không dump toàn bộ cơ sở dữ liệu vào prompt.
\end{itemize}

\textbf{Xử lý Bất đồng bộ}

\begin{itemize}[leftmargin=*, label={--}]
    \item \textbf{BullMQ} (dependency) để xử lý job nặng (email, notification)
    theo hàng đợi, tránh block event loop của Node.js.

    \item \texttt{triggerTicketEmail} được gọi theo kiểu fire-and-forget
    (không \texttt{await}) trong webhook handler để phản hồi PayOS nhanh
    nhất có thể.
\end{itemize}

\subsubsection{Độ tin cậy và Tính nhất quán Dữ liệu (Reliability \& Data Integrity)}

\textbf{Giao dịch Nguyên tử}

\begin{itemize}[leftmargin=*, label={--}]
    \item \textbf{MongoDB Transactions} trong cron job: Mỗi đơn hàng pending
    hết hạn được xử lý trong một session transaction riêng biệt. Nếu xảy ra
    lỗi thì \texttt{abortTransaction()} -- đảm bảo không có trạng thái nửa
    vời (vé đã thu hồi nhưng số lượng chưa được hoàn trả).
\end{itemize}

\textbf{Cơ chế Tự phục hồi}

\begin{itemize}[leftmargin=*, label={--}]
    \item \textbf{Cron Job mỗi 1 phút:} Tự động quét và thu hồi vé của đơn
    hàng có \texttt{paymentStatus: 'pending'} quá 15 phút -- giải phóng slot
    cho người mua tiếp theo.

    \item \textbf{ScheduledJob model} hỗ trợ \texttt{retryCount} /
    \texttt{maxRetries: 3} / \texttt{lastError} -- cho phép retry job thất
    bại (email, cleanup).

    \item \textbf{Graceful startup:} Nếu MongoDB không kết nối được thì
    \texttt{process.exit(1)} ngay lập tức, không để server chạy ở trạng
    thái lỗi không xác định.

    \item \textbf{DNS override} sang \texttt{8.8.8.8}: Tránh lỗi DNS
    resolver của Render khi kết nối MongoDB Atlas.
\end{itemize}

\textbf{Validation Dữ liệu}

\begin{itemize}[leftmargin=*, label={--}]
    \item \textbf{Mongoose validators:}
    \begin{itemize}[label={\(\circ\)}]
        \item \texttt{endDateTime} $\geq$ \texttt{startDateTime} (Event).
        \item \texttt{discountAmount} $\leq$ \texttt{maxAmount} (Voucher).
        \item \texttt{validUntil} $>$ \texttt{now()} (Voucher).
        \item Các ràng buộc \texttt{min} / \texttt{max} trên các trường số liệu.
    \end{itemize}

    \item \textbf{Pre-save / Pre-update hooks:} Tự động tính
    \texttt{avgPurchasePrice}, kiểm tra logic nghiệp vụ trước khi ghi vào
    cơ sở dữ liệu.

    \item \textbf{\texttt{runValidators: true}} trong
    \texttt{findByIdAndUpdate} đảm bảo Mongoose validator chạy ngay cả
    khi thực hiện update.
\end{itemize}

\subsubsection{Khả năng Mở rộng (Scalability)}

\begin{itemize}[leftmargin=*, label={--}]
    \item \textbf{Cloud-native deployment:} Backend deploy trên \textbf{Render}
    (lắng nghe \texttt{0.0.0.0}), Frontend deploy trên \textbf{Vercel}.

    \item \textbf{Cloudinary} làm CDN lưu ảnh: Không lưu file trên server,
    dễ dàng scale theo chiều ngang.

    \item \textbf{BullMQ + Redis} tách biệt xử lý job nặng khỏi HTTP request
    cycle -- sẵn sàng chạy worker process riêng biệt.

    \item \textbf{HuggingFace dataset} lưu data training ML -- pipeline AI/ML
    tách rời hoàn toàn khỏi backend chính.

    \item Kiến trúc \textbf{MVC tách biệt rõ ràng:}
    routes / controllers / services / models -- mỗi tầng có thể thay thế
    hoặc mở rộng độc lập mà không ảnh hưởng tầng khác.
\end{itemize}

\subsubsection{Khả năng Bảo trì (Maintainability)}

\begin{itemize}[leftmargin=*, label={--}]
    \item \textbf{ES Modules} (\texttt{"type": "module"}) toàn bộ backend --
    chuẩn modern JavaScript.

    \item \textbf{TypeScript} toàn bộ frontend -- type-safe, bắt lỗi tại
    compile-time thay vì runtime.

    \item \textbf{Nodemon} cho phép hot-reload trong môi trường phát triển.

    \item \textbf{Environment-based configuration:} Mọi secret và cấu hình
    đều thông qua file \texttt{.env}:
    \texttt{SECRET\_KEY}, \texttt{MONGO\_URI}, \texttt{PAYOS\_*},
    \texttt{CLOUDINARY\_*}, \texttt{GEMINI\_API\_KEY}, \texttt{HF\_TOKEN},
    \texttt{CRON\_SECRET}, \texttt{FRONTEND\_URL}.

\subsubsection{Riêng tư Dữ liệu (Privacy)}

\begin{itemize}[leftmargin=*, label={--}]
    \item Trường \texttt{password} sử dụng \texttt{select: false} --
    không bao giờ trả về trong response dù query trực tiếp vào collection.

    \item Trường \texttt{avgPurchasePrice} không cho phép client cập nhật
    trực tiếp -- field bị loại bỏ khỏi payload nếu có trong request body.

    \item Vị trí địa lý của người dùng lưu dạng GeoJSON -- chỉ phục vụ
    tính năng gợi ý sự kiện gần vị trí, không expose ra ngoài trừ khi
    người dùng xem thông tin profile của chính mình.
\end{itemize}
